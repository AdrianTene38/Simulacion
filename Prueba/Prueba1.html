def repairData(data,nombre):
    name = nombre + '.csv'
    data.to_csv(name,sep=',')
    df = pd.read_csv(name, skiprows=2, low_memory=False)
    df = df.drop(df[df['Sector Formal']=='-'].index)
    df= df.rename(columns = {'Unnamed: 1':'Fechas'})
    return df
